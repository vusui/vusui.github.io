BackTop返回顶部

    基础用法

      <div class="vus-backtop vi-arrow-back vus-anim fast vus-rotate-90" vus-back-top></div>

      <!--JS使用方法-->
<script>
   Vus.backtop('[vus-back-top]', { // DOM元素（默认[vus-back-top]）
      width: '40px',
      height: '40px',
      lineHeight: '40px',
      color: '#fff',
      fontSize: '18px',
      right: '30px',
      bottom: '110px',
      radius: '50%',
      bgColor: 'rgba(0, 0, 0, .5)'
   }, 800); // 滚动位置大于多少才显示返回顶部按钮（默认800）
</script>
